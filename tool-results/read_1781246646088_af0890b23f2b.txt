   220→function pickStarters(
   221→  metadata: ProjectMetadata | undefined,
   222→  t: TranslateFn,
   223→): StarterPrompt[] {
   224→  const kind = metadata?.kind;
   225→  if (kind === 'image') return IMAGE_STARTERS;
   226→  if (kind === 'video') {
   227→    return metadata?.videoModel === 'hyperframes-html'
   228→      ? VIDEO_HYPERFRAMES_STARTERS
   229→      : VIDEO_SEEDANCE_STARTERS;
   230→  }
   231→  if (kind === 'audio') return AUDIO_STARTERS;
   232→  return DEFAULT_STARTER_KEYS.map((entry) => ({
   233→    icon: entry.icon,
   234→    title: t(entry.titleKey),
   235→    tag: t(entry.tagKey),
   236→    prompt: t(entry.promptKey),
   237→  }));
   238→}

   246→function ImportedFolderArtifacts({
   247→  projectId,
   248→  files,
   249→  onOpenFile,
   250→  t,
   251→}: {
   252→  projectId: string | null;
   253→  files: ProjectFile[];
   254→  onOpenFile?: (name: string) => void;
   255→  t: TranslateFn;
   256→}) {
   257→  const [visibleCount, setVisibleCount] = useState(IMPORTED_ARTIFACTS_INITIAL_VISIBLE_COUNT);
   258→
   259→  useEffect(() => {
   260→    setVisibleCount(IMPORTED_ARTIFACTS_INITIAL_VISIBLE_COUNT);
   261→  }, [files]);
   262→
   263→  if (files.length === 0) {
   264→    return (
   265→      <div className="chat-design-artifacts-empty" data-testid="chat-design-artifacts-empty">
   266→        {t('designFiles.empty')}
   267→      </div>
   268→    );
   269→  }
   270→
   271→  const visibleFiles = files.slice(0, visibleCount);
   272→  const hiddenCount = Math.max(0, files.length - visibleFiles.length);
   273→  const revealCount = Math.min(IMPORTED_ARTIFACTS_REVEAL_COUNT, hiddenCount);
   274→  const revealLabel = t('chat.designArtifactsShowMore', { count: revealCount });
   275→
   276→  return (
   277→    <div className="chat-design-artifacts" data-testid="chat-design-artifacts">
   278→      {visibleFiles.map((file, index) => {
   279→        const openable = Boolean(onOpenFile);
   280→        const openLabel = `${t('designFiles.previewOpen')} ${file.name}`;
   281→        const openFile = () => {
   282→          onOpenFile?.(file.name);
   283→        };
   284→        return (
   285→          <div
   286→            key={file.name}
   287→            className="chat-design-artifact"
   288→            data-kind={file.kind}
   289→            data-file-name={file.name}
   290→            data-testid={`chat-design-artifact-${index}`}
   291→            role={openable ? 'button' : 'listitem'}
   292→            tabIndex={openable ? 0 : undefined}
   293→            title={openLabel}
   294→            aria-label={openLabel}
   295→            onDoubleClick={openable ? openFile : undefined}
   296→            onKeyDown={
   297→              openable
   298→                ? (event) => {
   299→                    if (event.key !== 'Enter' && event.key !== ' ') return;
   300→                    event.preventDefault();
   301→                    openFile();
   302→                  }
   303→                : undefined
   304→            }
   305→          >
   306→            <div className="chat-design-artifact-preview" aria-hidden>
   307→              <ChatArtifactPreview projectId={projectId} file={file} />
   308→            </div>
   309→            <div className="chat-design-artifact-meta">
   310→              <span className="chat-design-artifact-name" title={file.name}>
   311→                {file.name}
   312→              </span>
   313→              <span className="chat-design-artifact-kind">
   314→                {chatArtifactKindLabel(file.kind, t)}
   315→              </span>
   316→            </div>
   317→          </div>
   318→        );
   319→      })}
   320→      {hiddenCount > 0 ? (
   321→        <button
   322→          type="button"
   323→          className="chat-design-artifact chat-design-artifact-more"
   324→          data-testid="chat-design-artifacts-more"
   325→          aria-label={revealLabel}
   326→          title={revealLabel}
   327→          onClick={() => {
   328→            setVisibleCount((current) =>
   329→              Math.min(files.length, current + IMPORTED_ARTIFACTS_REVEAL_COUNT),
   330→            );
   331→          }}
   332→        >
   333→          <span className="chat-design-artifact-more-icon" aria-hidden>
   334→            +
   335→          </span>
   336→          <span className="chat-design-artifact-more-count">
   337→            {revealLabel}
   338→          </span>
   339→        </button>
   340→      ) : null}
   341→    </div>
   342→  );
   343→}

   345→function ChatArtifactPreview({
   346→  projectId,
   347→  file,
   348→}: {
   349→  projectId: string | null;
   350→  file: ProjectFile;
   351→}) {
   352→  if (!projectId) {
   353→    return <ChatArtifactFallback kind={file.kind} />;
   354→  }
   355→
   356→  const url = `${projectRawUrl(projectId, file.name)}?v=${Math.round(file.mtime)}`;
   357→  if (isRenderableSketchJson(file)) {
   358→    return <SketchPreview projectId={projectId} file={file} />;
   359→  }
   360→  if (file.kind === 'image' || file.kind === 'sketch') {
   361→    return <img src={url} alt="" loading="lazy" />;
   362→  }
   363→  if (file.kind === 'html') {
   364→    return (
   365→      <iframe
   366→        title={file.name}
   367→        src={url}
   368→        sandbox="allow-scripts allow-downloads"
   369→        loading="lazy"
   370→      />
   371→    );
   372→  }
   373→  if (file.kind === 'video') {
   374→    return <video src={url} muted playsInline preload="metadata" />;
   375→  }
   376→  return <ChatArtifactFallback kind={file.kind} />;
   377→}

   379→function ChatArtifactFallback({ kind }: { kind: ProjectFile['kind'] }) {
   380→  return (
   381→    <span className="chat-design-artifact-fallback">
   382→      <Icon name={chatArtifactIcon(kind)} size={28} />
   383→      <span>{chatArtifactShortKind(kind)}</span>
   384→    </span>
   385→  );
   386→}
   387→
   388→function chatArtifactIcon(kind: ProjectFile['kind']): IconName {
   389→  if (kind === 'html' || kind === 'code') return 'file-code';
   390→  if (kind === 'image' || kind === 'sketch') return 'image';
   391→  if (kind === 'video' || kind === 'audio') return 'play';
   392→  if (kind === 'presentation') return 'present';
   393→  return 'file';
   394→}
   395→
   396→function chatArtifactShortKind(kind: ProjectFile['kind']): string {
   397→  if (kind === 'html') return 'HTML';
   398→  if (kind === 'image') return 'IMG';
   399→  if (kind === 'sketch') return 'SKETCH';
   400→  if (kind === 'video') return 'VIDEO';
   401→  if (kind === 'pdf') return 'PDF';
   402→  if (kind === 'presentation') return 'PPT';
   403→  if (kind === 'document') return 'DOC';
   404→  return 'FILE';
   405→}
   406→
   407→function chatArtifactKindLabel(kind: ProjectFile['kind'], t: TranslateFn): string {
   408→  if (kind === 'html') return t('designFiles.kindHtml');
   409→  if (kind === 'image') return t('designFiles.kindImage');
   410→  if (kind === 'sketch') return t('designFiles.kindSketch');
   411→  if (kind === 'video') return 'Video';
   412→  if (kind === 'audio') return 'Audio';
   413→  if (kind === 'pdf') return t('designFiles.kindPdf');
   414→  if (kind === 'document') return t('designFiles.kindDocument');
   415→  if (kind === 'presentation') return t('designFiles.kindPresentation');
   416→  if (kind === 'spreadsheet') return t('designFiles.kindSpreadsheet');
   417→  return t('designFiles.kindBinary');
   418→}
