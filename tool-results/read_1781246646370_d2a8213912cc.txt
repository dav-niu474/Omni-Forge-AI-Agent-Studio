   233→function mergeServerMessageWithLocal(server: ChatMessage, local?: ChatMessage): ChatMessage {
   234→  if (!local) return server;
   235→  const merged: ChatMessage = { ...server };
   236→  if (!server.producedFiles?.length && local.producedFiles?.length) {
   237→    merged.producedFiles = local.producedFiles;
   238→  }
   239→  if (!server.preTurnFileNames?.length && local.preTurnFileNames?.length) {
   240→    merged.preTurnFileNames = local.preTurnFileNames;
   241→  }
   242→  if (!server.lastRunEventId && local.lastRunEventId) {
   243→    merged.lastRunEventId = local.lastRunEventId;
   244→  }
   245→  if (!server.startedAt && local.startedAt) {
   246→    merged.startedAt = local.startedAt;
   247→  }
   248→  if (!server.endedAt && local.endedAt) {
   249→    merged.endedAt = local.endedAt;
   250→  }
   251→  if (!server.runStatus && local.runStatus) {
   252→    merged.runStatus = local.runStatus;
   253→  }
   254→  return merged;
   255→}

   257→export function mergeServerMessagesIntoConversation(
   258→  current: ChatMessage[],
   259→  serverMessages: ChatMessage[],
   260→): ChatMessage[] {
   261→  const currentById = new Map(current.map((message) => [message.id, message]));
   262→  const serverIds = new Set(serverMessages.map((message) => message.id));
   263→  const merged = serverMessages.map((message) =>
   264→    mergeServerMessageWithLocal(message, currentById.get(message.id)),
   265→  );
   266→  for (const message of current) {
   267→    if (!serverIds.has(message.id)) merged.push(message);
   268→  }
   269→  return merged;
   270→}

   375→function maxChatPanelWidthForSplit(splitWidth: number): number {
   376→  if (!Number.isFinite(splitWidth) || splitWidth <= 0) return MAX_CHAT_PANEL_WIDTH;
   377→  const workspaceMinWidth = workspacePanelMinWidthForSplit(splitWidth);
   378→  const viewportAwareMax = splitWidth - SPLIT_RESIZE_HANDLE_WIDTH - workspaceMinWidth;

   442→function historyWithWorkspaceContext(

   578→    const key = designSystemAuditAutoRepairKey(projectId);

   588→    } else {
   589→      window.sessionStorage.removeItem(key);
   590→    }
   591→    return true;
   592→  } catch {
   593→    return false;
   594→  }
   595→}

   720→// The media model the user picked in the New Project → Media dialog, keyed by
   721→// surface. For BYOK providers (AIHubMix) media is produced by the generate_*
   722→// chat tools whose default model comes from the per-request byok*Model field —
   723→// NOT the `od media generate` dispatcher — so without this seed the dialog pick

   726→// back to the Settings default exactly as before. The daemon re-validates the id

   728→function projectMediaModelSeed(
   729→  metadata: ProjectMetadata | null | undefined,
   730→  surface: 'image' | 'video' | 'speech',
   731→): string | undefined {
   732→  if (!metadata) return undefined;

   734→    return metadata.imageModel?.trim() || undefined;
   735→  }
   736→  if (surface === 'video' && metadata.kind === 'video') {
   737→    return metadata.videoModel?.trim() || undefined;
   738→  }
   739→  if (surface === 'speech' && metadata.kind === 'audio' && metadata.audioKind === 'speech') {
   740→    return metadata.audioModel?.trim() || undefined;
   741→  }
   742→  return undefined;
   743→}
