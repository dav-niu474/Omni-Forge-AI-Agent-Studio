    28→// Drag Hook
    29→// ============================================================================
    30→function useDrag(initialPos: { x: number; y: number }) {
    31→  const [pos, setPos] = useState(initialPos);
    32→  const [isDragging, setIsDragging] = useState(false);
    33→  const dragRef = useRef<{ startX: number; startY: number; origX: number; origY: number } | null>(null);
    34→
    35→  const onMouseDown = useCallback((e: React.MouseEvent) => {
    36→    // Only drag from header area
    37→    if ((e.target as HTMLElement).closest("[data-drag-handle]")) {
    38→      e.preventDefault();
    39→      dragRef.current = {
    40→        startX: e.clientX, startY: e.clientY,
    41→        origX: pos.x, origY: pos.y,
    42→      };
    43→      setIsDragging(true);
    44→    }
    45→  }, [pos]);
    46→
    47→  useEffect(() => {
    48→    if (!isDragging) return;
    49→    const onMouseMove = (e: MouseEvent) => {
    50→      if (!dragRef.current) return;
    51→      const dx = e.clientX - dragRef.current.startX;
    52→      const dy = e.clientY - dragRef.current.startY;
    53→      setPos({
    54→        x: dragRef.current.origX + dx,
    55→        y: dragRef.current.origY + dy,
    56→      });
    57→    };
    58→    const onMouseUp = () => {
    59→      setIsDragging(false);
    60→      dragRef.current = null;
    61→    };
    62→    window.addEventListener("mousemove", onMouseMove);
    63→    window.addEventListener("mouseup", onMouseUp);
    64→    return () => {
    65→      window.removeEventListener("mousemove", onMouseMove);
    66→      window.removeEventListener("mouseup", onMouseUp);
    67→    };
    68→  }, [isDragging]);
    69→
    70→  return { pos, isDragging, onMouseDown, setPos };
    71→}

   191→// ============================================================================
   192→// TRUE Floating Card — absolutely positioned, draggable, glass effect
   193→// ============================================================================
   194→function FloatingCard({
   195→  children, visible, onClose, title, icon: Icon, accentColor,
   196→  defaultPos, defaultSize, zIndex, onBringToFront, className,
   197→}: {
   198→  children: React.ReactNode;
   199→  visible: boolean;
   200→  onClose?: () => void;
   201→  title: string;
   202→  icon: React.ElementType;
   203→  accentColor?: string;
   204→  defaultPos: { x: number; y: number };
   205→  defaultSize: { w: number | string; h: number | string };
   206→  zIndex: number;
   207→  onBringToFront: () => void;
   208→  className?: string;
   209→}) {
   210→  const { pos, isDragging, onMouseDown, setPos } = useDrag(defaultPos);
   211→  const [size, setSize] = useState(defaultSize);
   212→  const [isMaximized, setIsMaximized] = useState(false);
   213→
   214→  // Reset position when visibility changes
   215→  useEffect(() => {
   216→    if (visible) {
   217→      setPos(defaultPos);
   218→      setSize(defaultSize);
   219→      setIsMaximized(false);
   220→    }
   221→  }, [visible]);
   222→
   223→  if (!visible) return null;
   224→
   225→  const cardStyle: React.CSSProperties = isMaximized
   226→    ? {
   227→        position: "absolute",
   228→        left: 12, top: 12, right: 12, bottom: 12,
   229→        zIndex,
   230→        background: "var(--card-bg)",
   231→        border: "1px solid var(--card-border)",
   232→        boxShadow: isDragging
   233→          ? "0 24px 80px rgba(0,0,0,0.5), 0 8px 24px rgba(0,0,0,0.3)"
   234→          : "var(--card-shadow)",
   235→        backdropFilter: "blur(24px)",
   236→        borderRadius: 20,
   237→        transition: isDragging ? "none" : "box-shadow 0.2s, left 0.3s, top 0.3s, right 0.3s, bottom 0.3s",
   238→      }
   239→    : {
   240→        position: "absolute",
   241→        left: pos.x,
   242→        top: pos.y,
   243→        width: typeof size.w === "number" ? size.w : undefined,
   244→        right: typeof size.w === "string" ? size.w : undefined,
   245→        height: typeof size.h === "number" ? size.h : undefined,
   246→        bottom: typeof size.h === "string" ? size.h : undefined,
   247→        zIndex,
   248→        background: "var(--card-bg)",
   249→        border: "1px solid var(--card-border)",
   250→        boxShadow: isDragging
   251→          ? "0 24px 80px rgba(0,0,0,0.5), 0 8px 24px rgba(0,0,0,0.3)"
   252→          : "var(--card-shadow)",
   253→        backdropFilter: "blur(24px)",
   254→        borderRadius: 20,
   255→        cursor: isDragging ? "grabbing" : "default",
   256→        transition: isDragging ? "none" : "box-shadow 0.2s",
   257→      };
   258→
   259→  return (
   260→    <AnimatePresence>
   261→      {visible && (
   262→        <motion.div
   263→          initial={{ opacity: 0, scale: 0.92, y: 16 }}
   264→          animate={{ opacity: 1, scale: 1, y: 0 }}
   265→          exit={{ opacity: 0, scale: 0.92, y: 16 }}
   266→          transition={{ duration: 0.25, ease: [0.23, 1, 0.32, 1] }}
   267→          className={cn("flex flex-col overflow-hidden", className)}
   268→          style={cardStyle}
   269→          onMouseDown={onBringToFront}
   270→        >
   271→          {/* Drag handle header */}
   272→          <div
   273→            data-drag-handle
   274→            onMouseDown={onMouseDown}
   275→            className="flex items-center justify-between px-4 h-11 shrink-0 select-none"
   276→            style={{ borderBottom: "1px solid var(--glass-border)", cursor: isDragging ? "grabbing" : "grab" }}
   277→          >
   278→            <div className="flex items-center gap-2.5">
   279→              <div
   280→                className="size-6 rounded-lg flex items-center justify-center"
   281→                style={{ background: "var(--accent-tint)" }}
   282→              >
   283→                <Icon className="size-3.5" style={{ color: accentColor || "var(--accent)" }} />
   284→              </div>
   285→              <span className="text-[13px] font-semibold" style={{ color: "var(--text-strong)" }}>{title}</span>
   286→            </div>
   287→            <div className="flex items-center gap-0.5">
   288→              <button
   289→                onClick={() => setIsMaximized(!isMaximized)}
   290→                className="size-7 rounded-lg flex items-center justify-center transition-colors"
   291→                style={{ color: "var(--text-faint)" }}
   292→                onMouseEnter={(e) => { e.currentTarget.style.background = "var(--bg-subtle)"; e.currentTarget.style.color = "var(--text-muted)"; }}
   293→                onMouseLeave={(e) => { e.currentTarget.style.background = "transparent"; e.currentTarget.style.color = "var(--text-faint)"; }}
   294→              >
   295→                {isMaximized ? <Minimize2 className="size-3.5" /> : <Maximize2 className="size-3.5" />}
   296→              </button>
   297→              {onClose && (
   298→                <button
   299→                  onClick={onClose}
   300→                  className="size-7 rounded-lg flex items-center justify-center transition-colors"
   301→                  style={{ color: "var(--text-faint)" }}
   302→                  onMouseEnter={(e) => { e.currentTarget.style.background = "rgba(239,68,68,0.15)"; e.currentTarget.style.color = "#ef4444"; }}
   303→                  onMouseLeave={(e) => { e.currentTarget.style.background = "transparent"; e.currentTarget.style.color = "var(--text-faint)"; }}
   304→                >
   305→                  <X className="size-3.5" />
   306→                </button>
   307→              )}
   308→            </div>
   309→          </div>
   310→          {children}
   311→        </motion.div>
   312→      )}
   313→    </AnimatePresence>
   314→  );
   315→}

   320→const QUICK_PROMPTS: Record<Modality, string[]> = {
   321→  text: ["Write a product description", "Create a blog outline", "Draft an email campaign"],
   322→  image: ["Design a logo concept", "Create a social media post", "Generate a hero illustration"],

   328→function ChatCardContent() {
   329→  const { messages, isStreaming, activeModality } = useStudioStore();
   330→  const [inputValue, setInputValue] = useState("");
   331→  const scrollRef = useRef<HTMLDivElement>(null);
   332→  const textareaRef = useRef<HTMLTextAreaElement>(null);
   333→  const modalityConfig = MODALITY_CONFIG[activeModality];
   334→  const hasValue = inputValue.trim().length > 0;
   335→
   336→  const handleSend = useCallback(() => {
   337→    const trimmed = inputValue.trim();
   338→    if (!trimmed || isStreaming) return;
   339→    const store = useStudioStore.getState();
   340→    store.addMessage({
   341→      id: `msg-${Date.now()}`,
   342→      role: "user",
   343→      content: trimmed,
   344→      modality: activeModality,
   345→      artifacts: [],
   346→      timestamp: Date.now(),
   347→    });
   348→    setInputValue("");
   349→    if (textareaRef.current) textareaRef.current.style.height = "auto";
   350→  }, [inputValue, isStreaming, activeModality]);
   351→
   352→  const handleKeyDown = useCallback((e: React.KeyboardEvent<HTMLTextAreaElement>) => {
   353→    if (e.key === "Enter" && !e.shiftKey) { e.preventDefault(); handleSend(); }
   354→  }, [handleSend]);
   355→
   356→  const handleTextareaChange = useCallback((e: React.ChangeEvent<HTMLTextAreaElement>) => {
   357→    setInputValue(e.target.value);
   358→    e.target.style.height = "auto";
   359→    e.target.style.height = Math.min(e.target.scrollHeight, 120) + "px";
   360→  }, []);
   361→
   362→  useEffect(() => {
   363→    if (scrollRef.current) scrollRef.current.scrollTop = scrollRef.current.scrollHeight;
   364→  }, [messages]);
   365→
   366→  const prompts = QUICK_PROMPTS[activeModality];
   367→
   368→  return (
   369→    <div className="flex flex-col flex-1 min-h-0">
   370→      <ScrollArea className="flex-1 studio-scrollbar">
   371→        <div ref={scrollRef} className="flex flex-col gap-4 p-4">
   372→          {messages.length === 0 && (
   373→            <div className="flex flex-col items-center justify-center py-6 px-4">
   374→              <div className="size-12 rounded-2xl flex items-center justify-center mb-3"
   375→                style={{ background: "var(--accent-tint)", boxShadow: `0 0 24px var(--dot-glow)` }}>
   376→                <Lightbulb className="size-5" style={{ color: "var(--accent)" }} />
   377→              </div>
   378→              <h3 className="text-[13px] font-semibold mb-1" style={{ color: "var(--text-strong)" }}>
   379→                What would you like to create?
   380→              </h3>
   381→              <p className="text-[11px] text-center mb-4" style={{ color: "var(--text-muted)" }}>
   382→                {modalityConfig.label} mode
   383→              </p>
   384→              <div className="flex flex-col gap-1 w-full max-w-[240px]">
   385→                {prompts.map((prompt) => (
   386→                  <button key={prompt} onClick={() => setInputValue(prompt)}
   387→                    className="flex items-center gap-2 px-2.5 py-1.5 rounded-lg text-[11px] text-left transition-all"
   388→                    style={{ color: "var(--text-muted)" }}
   389→                    onMouseEnter={(e) => { e.currentTarget.style.background = "var(--bg-subtle)"; e.currentTarget.style.color = "var(--text-strong)"; }}
   390→                    onMouseLeave={(e) => { e.currentTarget.style.background = "transparent"; e.currentTarget.style.color = "var(--text-muted)"; }}
   391→                  >
   392→                    <Wand2 className="size-3 shrink-0" style={{ color: "var(--text-faint)" }} />
   393→                    <span>{prompt}</span>
   394→                  </button>
   395→                ))}
   396→              </div>
   397→            </div>
   398→          )}
   399→          {messages.map((msg) => (
   400→            <MessageBubble key={msg.id} message={msg} />
   401→          ))}
   402→        </div>
   403→      </ScrollArea>
   404→
   405→      {/* Composer */}
   406→      <div className="p-3 pt-0">
   407→        <div className="flex flex-col rounded-xl transition-all"
   408→          style={{ background: "var(--bg-panel)", border: "1px solid var(--border)" }}>
   409→          <div className="px-3 pt-2.5">
   410→            <Textarea ref={textareaRef} value={inputValue} onChange={handleTextareaChange}
   411→              onKeyDown={handleKeyDown} placeholder="Describe what you want to create..."
   412→              className="min-h-[22px] max-h-[120px] resize-none border-0 bg-transparent p-0 text-[13px] shadow-none focus-visible:ring-0 focus-visible:outline-none"
   413→              style={{ color: "var(--text-color)" }} rows={1} disabled={isStreaming} />
   414→          </div>
   415→          <div className="flex items-center justify-between px-2 py-1.5">
   416→            <div className="flex items-center gap-1 px-1.5 py-0.5 rounded-md text-[10px] font-medium"
   417→              style={{ background: "var(--accent-tint)", color: "var(--accent)" }}>
   418→              {modalityConfig.label}
   419→            </div>
   420→            <button onClick={handleSend} disabled={!hasValue}
   421→              className="size-7 rounded-lg flex items-center justify-center transition-all"
   422→              style={{
   423→                background: hasValue ? "var(--send-bg)" : "var(--bg-muted)",
   424→                color: hasValue ? "var(--send-fg)" : "var(--text-faint)",

   426→              <ArrowUp className="size-3.5" />
   427→            </button>
   428→          </div>

   476→              <div className="size-9 rounded-xl flex items-center justify-center" style={{ background: "var(--bg-subtle)" }}>
   477→                <HintIcon className="size-4" style={{ color: "var(--text-faint)" }} />
   478→              </div>
   479→            </div>
   480→          ))}
   481→        </div>
   482→      </div>
   483→    );
   484→  }
   485→
   486→  return (
   487→    <div className="flex flex-col flex-1 min-h-0">
   488→      <div className="flex items-center justify-between px-3 h-8 shrink-0"
   489→        style={{ borderBottom: "1px solid var(--glass-border)", background: "var(--bg-subtle)" }}>
   490→        <div className="flex items-center gap-2">
   491→          <div className="flex gap-1.5">

   496→          <span className="text-[11px]" style={{ color: "var(--text-soft)" }}>{activeArtifact.title}</span>
   497→        </div>
   498→      </div>

   504→        ) : activeArtifact.type === "image" ? (
   505→          <img src={activeArtifact.base64 ? `data:image/png;base64,${activeArtifact.base64}` : activeArtifact.url || ""}
   506→            alt={activeArtifact.alt || activeArtifact.title}
   507→            className="max-w-[92%] max-h-[90%] object-contain rounded-lg"
   508→            style={{ boxShadow: "0 4px 24px rgba(0,0,0,0.2)" }} />
   509→        ) : (
   510→          <div className="flex flex-col items-center gap-2" style={{ color: "var(--text-faint)" }}>
   511→            {activeArtifact.type === "video" && <Video className="size-10" />}
   512→            {activeArtifact.type === "audio" && <Music className="size-10" />}
   513→            {activeArtifact.type === "model3d" && <Box className="size-10" />}
   514→            {activeArtifact.type === "code" && <FileCode className="size-10" />}
   515→            <span className="text-[11px]">Preview: {activeArtifact.title}</span>
   516→          </div>
   517→        )}
   518→      </div>
   519→    </div>
   520→  );
   521→}

   523→// ============================================================================
   524→// Critique Card Content
   525→// ============================================================================
   526→function CritiqueCardContent() {
   527→  const { critiqueRounds } = useStudioStore();
   528→  const hasRounds = critiqueRounds.length > 0;
   529→
   530→  const ROLE_ICONS: Record<CritiqueRole, React.ElementType> = {
   531→    designer: Palette, critic: MessageSquare, brand: Shield,
   532→    a11y: Eye, copy: PenTool, modalist: Layers,
   533→  };
   534→  const VERDICT_CONFIG: Record<CritiqueVerdict, { label: string; color: string }> = {
   535→    ship: { label: "Ship", color: "#4caf72" },
   536→    degrade: { label: "Degrade", color: "#e09a40" },
   537→    fail: { label: "Fail", color: "#e06b65" },
   538→  };
   539→
   540→  if (!hasRounds) {
   541→    return (
   542→      <div className="flex flex-col items-center justify-center flex-1 gap-2 py-6">
   543→        <Sparkles className="size-5" style={{ color: "var(--accent)", opacity: 0.3 }} />
   544→        <p className="text-[11px]" style={{ color: "var(--text-faint)" }}>Critique scores appear after generation</p>
   545→      </div>
   546→    );
   547→  }
   548→
   549→  return (
   550→    <ScrollArea className="flex-1 studio-scrollbar">
   551→      <div className="flex flex-col gap-2 p-3">
   552→        {critiqueRounds.map((round) => {
   553→          const verdict = round.verdict ? VERDICT_CONFIG[round.verdict] : null;
   554→          return (
   555→            <div key={round.id} className="pl-3 py-1.5" style={{ borderLeft: "2px solid var(--accent)" }}>
   556→              <div className="flex items-center justify-between mb-1">
   557→                <span className="text-[10px] font-medium" style={{ color: "var(--text-muted)" }}>Round {round.number}</span>
   558→                <div className="flex items-center gap-1.5">
   559→                  {verdict && <span className="text-[9px] font-medium" style={{ color: verdict.color }}>{verdict.label}</span>}
   560→                  <span className="text-[12px] font-medium tabular-nums"
   561→                    style={{ color: round.overallScore >= 80 ? "#4caf72" : round.overallScore >= 60 ? "#e09a40" : "#e06b65" }}>
   562→                    {round.overallScore.toFixed(0)}
   563→                  </span>
   564→                </div>
   565→              </div>
   566→              {(Object.entries(round.roleScores) as [CritiqueRole, number][]).map(([role, score]) => {
   567→                const RoleIcon = ROLE_ICONS[role];
   568→                const roleConfig = CRITIQUE_ROLE_CONFIG[role];
   569→                return (
   570→                  <div key={role} className="flex items-center gap-2 py-0.5">
   571→                    <RoleIcon className="size-2.5 shrink-0" style={{ color: "var(--text-faint)" }} />
   572→                    <span className="text-[10px] flex-1" style={{ color: "var(--text-muted)" }}>{roleConfig.label}</span>
   573→                    <div className="w-12 h-1 rounded-full overflow-hidden" style={{ background: "var(--bg-muted)" }}>
   574→                      <div className="h-full rounded-full" style={{
   575→                        width: `${score}%`,
   576→                        background: score >= 80 ? "#4caf72" : score >= 60 ? "#e09a40" : "#e06b65",
   577→                        opacity: 0.6,
   578→                      }} />
   579→                    </div>
   580→                    <span className="text-[10px] tabular-nums w-5 text-right" style={{ color: "var(--text-soft)" }}>{score.toFixed(0)}</span>
   581→                  </div>
   582→                );
   583→              })}
   584→            </div>
   585→          );
   586→        })}
   587→      </div>
   588→    </ScrollArea>
   589→  );
   590→}
